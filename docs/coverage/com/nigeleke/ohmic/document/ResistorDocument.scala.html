45 object ResistorDocument:
46 
47   val sheetName = "Resistor Codes"
48 
49   def apply(): ResistorDocument = ResistorDocument(XSSFWorkbook()).withSheet().withHeader()
50 
51   extension (d: ResistorDocument)
52     def withSheet() =
53       d.document.createSheet(sheetName)
54       d
55 
56     def withHeader(): ResistorDocument =
57       val row = nextRow()
58       row.createCell(0).setCellValue("Value")
59       row.createCell(1).setCellValue("Unit")
60       row.createCell(2).setCellValue("Colours")
61       d
62 
63     def nextRow() =
64       val sheet    = d.document.getSheet(sheetName)
65       val rowCount = sheet.getPhysicalNumberOfRows
66       sheet.createRow(rowCount)
67 
68     def saveAs(filePath: Path) =
69       val out = FileOutputStream(filePath.toFile)
70       d.document.write(out)
71       out.close()
72       d
73 
74     def withRow(resistorRow: ResistorRow): ResistorDocument =
75       val row = nextRow()
76       row.createCell(0).setCellValue(resistorRow.baseValue)
77       row.createCell(1).setCellValue(resistorRow.symbol)
78 
79       val colours = resistorRow.colours
80         .filter(_.size != 0)
81         .sortBy(_.size)
82         .map(_.mkString(" - "))
83         .mkString("\n")
84 
85       row.createCell(2).setCellValue(colours)
86       d
87 
88     def withRows(rows: Seq[ResistorRow]) = rows.foldLeft(d)((d0, r) => d0.withRow(r))
