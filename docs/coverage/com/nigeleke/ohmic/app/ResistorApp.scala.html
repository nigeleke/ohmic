43   private val series = Resistor.setFor(Series.e24)
44 
45   val series5BandMap = (Resistor.fullSet
46     .filter(series.contains)
47     .filter(_.colourCodes.size == 4)
48     .map(r => (r.value, r)))
49     .toMap
50 
51   val rows = for
52     r1 <- series.toList.sortBy(_.value)
53     optR2 = series5BandMap.get(r1.value)
54   yield
55     val v = r1.formattedValue
56     val colourCodes =
57       Seq(r1.colourCodes) :+ (optR2.map(_.colourCodes).getOrElse(Seq.empty[BandColour]))
58     ResistorRow(v.value, v.unit.symbol, colourCodes)
59 
60   ResistorDocument()
61     .withRows(rows.toSeq)
62     .saveAs(Paths.get("resistors.xlsx"))
